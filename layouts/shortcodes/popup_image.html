".") -}} {{- errorf "Image caption ends with a period (%s)" .Position -}} {{- end -}} {{- end -}} {{- if not $width -}} {{- $width = "100%" -}} {{- end -}} {{- if not $imageid -}} {{- $imageid = "lightbox" -}} {{- end -}} {{- /* Turn relative values for $link into absolute URLs */ -}} {{- $prefix := slicestr $link 0 1 -}} {{- if (ne $prefix "/") -}} {{- $prefix := slicestr $link 0 8 -}} {{- if (ne $prefix "https://") -}} {{- $prefix := slicestr $link 0 3 -}} {{- if (eq $prefix "../") -}} {{- $link = printf "/%s%s" .Page.Dir (slicestr $link 3) -}} {{- else -}} {{- $link = printf "/%s%s" .Page.Dir $link -}} {{- end -}} {{- end -}} {{- end -}} {{- $path := printf "content/%s" $link }} {{- if strings.HasSuffix $link ".svg" -}} {{- $file := readFile $path -}} {{- $viewBox := index (findRE "viewBox( *?)=( *?)\"(.*?)\"" $file 1) 0 -}} {{- $viewBox := index (findRE "[0123456789 \\.]+" $viewBox 1) 0 -}} {{- $sizes := split $viewBox " " -}} {{- $w := sub (float (index $sizes 2)) (float (index $sizes 0)) -}} {{- $h := sub (float (index $sizes 3)) (float (index $sizes 1)) -}} {{- $ratio = mul (div $h $w) 100 -}} {{- else -}} {{- if not $ratio -}} {{- $ratio := 1 -}} {{- end -}} {{- end -}} {{- /* Ensure that $ratio does NOT end with % */ -}} {{- $ratio = strings.TrimRight "%" $ratio -}}
